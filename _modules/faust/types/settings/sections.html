    """Configuration section."""

    type: SectionType
    title: str
    refid: str
    content: Optional[str]

    def __init__(
        self,
        type: SectionType,
        title: str,
        refid: str,
        *,
        content: Optional[str] = None,
    ) -> None:
        self.type = type
        self.title = title
        self.refid = refid
        self.content = content
        SECTIONS[self.type] = self


[docs]    def setting(
        self, param: Type[_Param[IT, OT]], **kwargs: Any
    ) -> Callable[[Callable], OT]:
        """Decorate to define new setting in this section."""

        def inner(fun: Callable) -> OT:
            setting = param(
                name=fun.__name__,
                section=self,
                help=fun.__doc__,
                **kwargs,
            )
            return cast(OT, setting)

        return inner


    def __repr__(self) -> str:
        return f"<{type(self).__name__}: {self.type}>"
